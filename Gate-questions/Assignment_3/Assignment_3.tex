	 \item A square matrix $\vec{B}$ is skew-symmetric if\hfill{(2009-CE)}

 \begin{enumerate}
  \item $\vec{B}^{\top} = - \vec{B}$
  \item $\vec{B}^{\top} = \vec{B}$
  \item $\vec{B}^{-1} =  \vec{B}$
  \item $\vec{B}^{-1} =  \vec{B}^{T}$
 \end{enumerate}
